        mat = new Material(Shader.Find("Standard"));
        m_SceneView = new SceneView();

    enum MotionState
    {
        kInactive,
        kActive,
        kDragging
    }

    public enum ViewTool
    {
        None = -1,
        Orbit = 0,
        Pan = 1,
        Zoom = 2,
        FPS = 3
    }

    enum DraggingLockedState
    {
        NotDragging, // Default state. Scene view camera is snapped to selected object instantly
        Dragging, // User is dragging from handles. Scene view camera holds still.
        LookAt // Temporary state after dragging or selection change, where we return scene view camera smoothly to selected object
    }

    public class GUIEvent
    {
        public EventType type;
        public int button;
        public Vector2 delta;
        public Vector2 mousePosition;
        public bool alt;
        public bool shift;
        public KeyCode keyCode;
    }

    class SceneView
    {
        const float k_MaxSceneViewSize = 3.2e38f;
        const float kDefaultViewSize = 10f;
        const float s_FPSScrollWheelMultiplier = 0.01f;
        const float k_FlySpeedAcceleration = 1.8f;

        public Camera camera;
        public Rect viewRect;

        private AnimFloat m_Size = new AnimFloat(kDefaultViewSize);
        private AnimBool m_Ortho = new AnimBool();
        private AnimVector3 m_Position = new AnimVector3(new Vector3(0, 0, 0));
        private AnimQuaternion m_Rotation = new AnimQuaternion(Quaternion.identity);
        private object m_LastLockedObject;
        private bool m_ViewIsLockedToObject;
        private bool isRotationLocked;
        private bool m_2DMode;

        private float s_StartZoom = 0f;
        private float s_ZoomSpeed = 0f;
        private float s_TotalMotion = 0f;
        private Vector3 s_Motion;
        private bool s_Moving;
        private float k_FlySpeed = 9f;
        private float s_FlySpeedTarget = 0f;
        private AnimVector3 s_FlySpeed = new AnimVector3(Vector3.zero);
        private CameraFlyModeContext s_CameraFlyModeContext = new CameraFlyModeContext();

        private bool viewIsLockedToObject
        {
            get => m_ViewIsLockedToObject;
            set
            {
                if (value) m_LastLockedObject = new object();
                else m_LastLockedObject = null;

                m_ViewIsLockedToObject = value;
                draggingLocked = DraggingLockedState.LookAt;
            }
        }

        private bool in2DMode
        {
            get => m_2DMode;
            set
            {
                if (m_2DMode != value && viewTool != ViewTool.FPS && viewTool != ViewTool.Orbit)
                {
                    m_2DMode = value;
                    // On2DModeChange();
                }
            }
        }

        private int s_ButtonDown = -1;
        private DraggingLockedState draggingLocked;
        private ViewTool m_ViewTool = ViewTool.Pan;
        private ViewTool s_LockedViewTool = ViewTool.None;
        private MotionState s_CurrentState;

        private bool viewToolActive
        {
            get
            {
                if (GUIUtility.hotControl != 0 && s_LockedViewTool == ViewTool.None)
                    return false;

                return s_LockedViewTool != ViewTool.None || Event.current.alt || (s_ButtonDown == 1) || (s_ButtonDown == 2);
            }
        }

        public ViewTool viewTool
        {
            get
            {
                var e = guiEvent;
                if (e != null && viewToolActive)
                {
                    if (s_LockedViewTool == ViewTool.None)
                    {
                        var actionKey = EditorGUI.actionKey;
                        var noModifiers = !actionKey && !e.alt;

                        if ((s_ButtonDown <= 0 && noModifiers)
                            || (s_ButtonDown <= 0 && actionKey)
                            || s_ButtonDown == 2
                            || (in2DMode || isRotationLocked) && !(s_ButtonDown == 1 && e.alt)
                           )
                        {
                            m_ViewTool = ViewTool.Pan;
                        }
                        else if (s_ButtonDown == 1 && e.alt)
                        {
                            m_ViewTool = ViewTool.Zoom;
                        }
                        else if (s_ButtonDown <= 0 && e.alt)
                        {
                            m_ViewTool = ViewTool.Orbit;
                        }
                        else if (s_ButtonDown == 1 && !e.alt)
                        {
                            m_ViewTool = ViewTool.FPS;
                        }
                    }
                }
                else
                {
                    m_ViewTool = ViewTool.Pan;
                }
                return m_ViewTool;
            }
            set => m_ViewTool = value;
        }

        static float ValidateSceneSize(float value)
        {
            if (value == 0f)
                return float.Epsilon;
            if (value > k_MaxSceneViewSize)
                return k_MaxSceneViewSize;
            if (value < -k_MaxSceneViewSize)
                return -k_MaxSceneViewSize;
            return value;
        }

        float perspectiveFov => camera.fieldOfView;

        public float size
        {
            get => m_Size.value;
            set => m_Size.value = ValidateSceneSize(value);
        }
        public Vector3 pivot
        {
            get => m_Position.value;
            set => m_Position.value = value;
        }

        // The direction of the scene view.
        public Quaternion rotation
        {
            get => m_Rotation.value;
            set => m_Rotation.value = value;
        }

        public float cameraDistance
        {
            get
            {
                float res;
                if (!camera.orthographic)
                {
                    var fov = m_Ortho.Fade(perspectiveFov, 0);
                    res = GetPerspectiveCameraDistance(size, fov);
                }
                else
                    res = size * 2f;

                // clamp to allowed range in case scene view size was huge
                return Mathf.Clamp(res, -k_MaxSceneViewSize, k_MaxSceneViewSize);
            }
        }

        public bool orthographic
        {
            get => m_Ortho.value;
            set
            {
                m_Ortho.value = value;
                // svRot.UpdateGizmoLabel(this, m_Rotation.target * Vector3.forward, m_Ortho.target);
            }
        }

        static float GetPerspectiveCameraDistance(float objectSize, float fov)
        {
            //        A
            //        |\        We want to place camera at a
            //        | \       distance that, at the given FOV,
            //        |  \      would enclose a sphere of radius
            //     _..+.._\     "size". Here |BC|=size, and we
            //   .'   |   '\    need to find |AB|. ACB is a right
            //  /     |    _C   angle, andBAC is half the FOV. So
            // |      | _-   |  that gives: sin(BAC)=|BC|/|AB|,
            // |      B      |  and thus |AB|=|BC|/sin(BAC).
            // |             |
            //  \           /
            //   '._     _.'
            //      `````
            return objectSize / Mathf.Sin(fov * 0.5f * Mathf.Deg2Rad);
        }

        private GUIEvent guiEvent;

        public void UpdateCamera()
        {
            var evt = Event.current;
            var e = new GUIEvent();
            e.type = evt.type;
            e.button = evt.button;
            e.mousePosition = evt.mousePosition;
            e.delta = evt.delta;
            e.alt = evt.alt;
            e.shift = evt.shift;
            e.keyCode = evt.keyCode;
            guiEvent = e;

            HandleClickAndDragToFocus();
            HandleViewToolCursor();

            if (s_LockedViewTool == ViewTool.FPS)
            {
                FixNegativeSize();
            }
            using (var inputSamplingScope = new CameraFlyModeContext.InputSamplingScope(s_CameraFlyModeContext, s_LockedViewTool, guiEvent, null, orthographic))
            {
                if (inputSamplingScope.currentlyMoving)
                    viewIsLockedToObject = false;

                if (inputSamplingScope.inputVectorChanged)
                    s_FlySpeedTarget = 0f;

                s_Motion = inputSamplingScope.currentInputVector;
            }

            if (e.type == EventType.ScrollWheel)
            {
                HandleScrollWheel(in2DMode == e.alt);
            }
            else if (e.type == EventType.MouseDrag)
            {
                HandleMouseDrag();
            }
            else if (e.type == EventType.MouseDown)
            {
                HandleMouseDown();
            }
            else if (e.type == EventType.MouseUp)
            {
                HandleMouseUp(e.button);
            }
            else if (e.type == EventType.KeyDown)
            {
                HandleKeyDown();
            }
            else if (e.type == EventType.Layout)
            {
                pivot += rotation * GetMovementDirection();
            }
        }

        private Vector3 GetMovementDirection()
        {
            s_Moving = s_Motion.sqrMagnitude > 0f;
            var deltaTime = s_CameraFlyModeContext.deltaTime;
            var speedModifier = 1f; //s_SceneView.cameraSettings.speed;

            if (guiEvent.shift)
                speedModifier *= 5f;

            if (s_Moving)
            {
                var accelerationEnabled = false;
                if (accelerationEnabled)
                    s_FlySpeedTarget = s_FlySpeedTarget < Mathf.Epsilon ? k_FlySpeed : s_FlySpeedTarget * Mathf.Pow(k_FlySpeedAcceleration, deltaTime);
                else
                    s_FlySpeedTarget = k_FlySpeed;
            }
            else
            {
                s_FlySpeedTarget = 0f;
            }

            var easingEnabled = false;
            if (easingEnabled)
            {
                var easingDuration = 1f;
                s_FlySpeed.speed = 1f / easingDuration;
                s_FlySpeed.target = s_Motion.normalized * s_FlySpeedTarget * speedModifier;
            }
            else
            {
                s_FlySpeed.value = s_Motion.normalized * s_FlySpeedTarget * speedModifier;
            }

            return s_FlySpeed.value * deltaTime;
        }

        void HandleViewToolCursor()
        {
            if (!viewToolActive) return;

            var cursor = MouseCursor.Arrow;
            switch (viewTool)
            {
                case ViewTool.Pan:
                    cursor = MouseCursor.Pan;
                    break;
                case ViewTool.Orbit:
                    cursor = MouseCursor.Orbit;
                    break;
                case ViewTool.FPS:
                    cursor = MouseCursor.FPS;
                    break;
                case ViewTool.Zoom:
                    cursor = MouseCursor.Zoom;
                    break;
            }
            if (cursor != MouseCursor.Arrow)
            {
                // AddCursorRect(cameraRect, cursor);
            }
        }

        private void HandleClickAndDragToFocus()
        {
            var e = guiEvent;
            if (e.type == EventType.MouseDrag)
                draggingLocked = DraggingLockedState.Dragging;
            else if (draggingLocked == DraggingLockedState.Dragging)
                draggingLocked = DraggingLockedState.LookAt;

            if (e.type == EventType.MouseDown)
            {
                s_ButtonDown = e.button;
            }
            // this is necessary because FPS tool won't get is cleanup logic
            // executed if another control uses the Event (i.e OnSceneGUI) (case 777346)
            else if (e.type == EventType.MouseUp && s_ButtonDown == e.button)
            {
                s_ButtonDown = -1;
            }
        }

        private void HandleScrollWheel(bool zoomTowardsCenter)
        {
            var e = guiEvent;
            float targetSize;
            var zoomDelta = e.delta.y;
            if (!camera.orthographic)
            {
                const float deltaCutoff = .3f;
                var relativeDelta = Mathf.Abs(size) * zoomDelta * .015f;
                if (relativeDelta > 0 && relativeDelta < deltaCutoff)
                    relativeDelta = deltaCutoff;
                else if (relativeDelta < 0 && relativeDelta > -deltaCutoff)
                    relativeDelta = -deltaCutoff;

                targetSize = size + relativeDelta;
            }
            else
            {
                targetSize = Mathf.Abs(size) * (zoomDelta * .015f + 1.0f);
            }

            var initialDistance = cameraDistance;
            if (!(float.IsNaN(targetSize) || float.IsInfinity(targetSize)))
            {
                targetSize = Mathf.Min(k_MaxSceneViewSize, targetSize);
                size = targetSize;
            }

            if (!zoomTowardsCenter && Mathf.Abs(cameraDistance) < 1.0e7f)
            {
                var percentage = 1f - (cameraDistance / initialDistance);
                Handles.SetCamera(viewRect, camera);
                var mouseRay = HandleUtility.GUIPointToWorldRay(e.mousePosition);
                var mousePivot = mouseRay.origin + mouseRay.direction * cameraDistance;
                var pivotVector = mousePivot - pivot;

                pivot += pivotVector * percentage;
            }
        }

        private void HandleMouseDown()
        {
            s_CurrentState = MotionState.kInactive;
            if (viewToolActive)
            {
                var wantedViewTool = viewTool;

                // Check if we want to lock a view tool
                if (s_LockedViewTool != wantedViewTool)
                {
                    s_LockedViewTool = wantedViewTool;

                    // Set up zoom parameters
                    s_StartZoom = size;
                    s_ZoomSpeed = Mathf.Max(Mathf.Abs(s_StartZoom), .3f);
                    s_TotalMotion = 0;

                    // we're not dragging yet, but enter this state so we can cleanup correctly
                    s_CurrentState = MotionState.kActive;
                    // This prevents camera from moving back on mousedown... find out why... and write a comment
                    GUIUtility.ExitGUI();
                }
            }
        }

        private void HandleMouseUp(int button)
        {
            // Move pivot to clicked point.
            if (button == 2 && s_CurrentState != MotionState.kDragging)
            {
                if (RaycastWorld(guiEvent.mousePosition, out var hit))
                {
                    var targetSize = size;
                    if (!orthographic)
                    {
                        var currentPosition = pivot - rotation * Vector3.forward * cameraDistance;
                        targetSize = size * Vector3.Dot(hit.point - currentPosition, rotation * Vector3.forward) / cameraDistance;
                    }
                    LookAt(hit.point, rotation, targetSize);
                }
            }
            ResetDragState();
        }

        public void HandleMouseDrag()
        {
            var e = guiEvent;
            switch (s_LockedViewTool)
            {
                case ViewTool.Orbit:
                {
                    if (!in2DMode && !isRotationLocked)
                    {
                        OrbitCameraBehavior();
                    }
                    break;
                }
                case ViewTool.Pan:
                {
                    FixNegativeSize();
                    viewIsLockedToObject = false;
                    var screenPos = camera.WorldToScreenPoint(pivot);
                    screenPos += new Vector3(-e.delta.x, e.delta.y, 0);
                    var worldDelta = camera.ScreenToWorldPoint(screenPos) - pivot;
                    worldDelta *= EditorGUIUtility.pixelsPerPoint;
                    if (e.shift) worldDelta *= 4;
                    pivot += worldDelta;
                    break;
                }
                case ViewTool.Zoom:
                {
                    var zoomDelta = HandleUtility.niceMouseDeltaZoom * (e.shift ? 9 : 3);
                    if (orthographic)
                    {
                        size = Mathf.Max(.0001f, size * (1 + zoomDelta * .001f));
                    }
                    else
                    {
                        s_TotalMotion += zoomDelta;

                        if (s_TotalMotion < 0)
                            size = s_StartZoom * (1 + s_TotalMotion * .001f);
                        else
                            size = size + zoomDelta * s_ZoomSpeed * .003f;
                    }
                    break;
                }
                case ViewTool.FPS:
                {
                    if (!in2DMode && !isRotationLocked)
                    {
                        if (!orthographic)
                        {
                            viewIsLockedToObject = false;

                            // The reason we calculate the camera position from the pivot, rotation and distance,
                            // rather than just getting it from the camera transform is that the camera transform
                            // is the *output* of camera motion calculations. It shouldn't be input and output at the same time,
                            // otherwise we easily get accumulated error.
                            // We did get accumulated error before when we did this - the camera would continuously move slightly in FPS mode
                            // even when not holding down any arrow/ASDW keys or moving the mouse.
                            var camPos = pivot - rotation * Vector3.forward * cameraDistance;

                            // Normal FPS camera behavior
                            var rot = rotation;
                            rot = Quaternion.AngleAxis(e.delta.y * .003f * Mathf.Rad2Deg, rot * Vector3.right) * rot;
                            rot = Quaternion.AngleAxis(e.delta.x * .003f * Mathf.Rad2Deg, Vector3.up) * rot;
                            rotation = rot;

                            pivot = camPos + rot * Vector3.forward * cameraDistance;
                        }
                        else
                        {
                            // We want orbit behavior in orthograpic when using FPS
                            OrbitCameraBehavior();
                        }
                        // svRot.UpdateGizmoLabel(view, view.rotation * Vector3.forward, view.m_Ortho.target);
                    }
                    break;
                }
            }
        }

        private void HandleKeyDown()
        {
            if (guiEvent.keyCode == KeyCode.Escape)
            {
                ResetDragState();
            }
        }

        private void OrbitCameraBehavior()
        {
            FixNegativeSize();
            var e = guiEvent;
            var rot = m_Rotation.target;
            rot = Quaternion.AngleAxis(e.delta.y * .003f * Mathf.Rad2Deg, rot * Vector3.right) * rot;
            rot = Quaternion.AngleAxis(e.delta.x * .003f * Mathf.Rad2Deg, Vector3.up) * rot;
            if (size < 0)
            {
                size = 0;
                pivot = camera.transform.position;
            }
            rotation = rot;
        }

        public void FixNegativeSize()
        {
            if (size == 0f) size = float.Epsilon;
            if (size < 0)
            {
                var fov = perspectiveFov;
                var distance = GetPerspectiveCameraDistance(size, fov);
                var p = m_Position.value + rotation * new Vector3(0, 0, -distance);
                size = -size;
                distance = GetPerspectiveCameraDistance(size, fov);
                m_Position.value = p + rotation * new Vector3(0, 0, distance);
            }
        }

        private void ResetDragState()
        {
            s_CurrentState = MotionState.kInactive;
            s_LockedViewTool = ViewTool.None;
            s_Motion = Vector3.zero;
            s_ButtonDown = -1;
        }

        private void ResetMotion()
        {
            s_Motion = Vector3.zero;
            s_FlySpeed.value = Vector3.zero;
            s_Moving = false;
        }

        bool RaycastWorld(Vector2 position, out RaycastHit hit)
        {
            hit = default;
            return false;
        }

        public void LookAt(Vector3 point)
        {
            FixNegativeSize();
            m_Position.target = point;
        }

        // Look at a specific point from a given direction.
        public void LookAt(Vector3 point, Quaternion direction)
        {
            FixNegativeSize();
            m_Position.target = point;
            m_Rotation.target = direction;
            // Update name in the top-right handle
            // svRot.UpdateGizmoLabel(this, direction * Vector3.forward, m_Ortho.target);
        }

        // Look directly at a specific point from a given direction.
        public void LookAtDirect(Vector3 point, Quaternion direction)
        {
            FixNegativeSize();
            m_Position.value = point;
            m_Rotation.value = direction;
            // Update name in the top-right handle
            // svRot.UpdateGizmoLabel(this, direction * Vector3.forward, m_Ortho.target);
        }

        // Look at a specific point from a given direction with a given zoom level.
        public void LookAt(Vector3 point, Quaternion direction, float newSize)
        {
            FixNegativeSize();
            m_Position.target = point;
            m_Rotation.target = direction;
            m_Size.target = Mathf.Abs(newSize);
            // Update name in the top-right handle
            // svRot.UpdateGizmoLabel(this, direction * Vector3.forward, m_Ortho.target);
        }

        // Look directally at a specific point from a given direction with a given zoom level.
        public void LookAtDirect(Vector3 point, Quaternion direction, float newSize)
        {
            FixNegativeSize();
            m_Position.value = point;
            m_Rotation.value = direction;
            size = Mathf.Abs(newSize);
            // Update name in the top-right handle
            // svRot.UpdateGizmoLabel(this, direction * Vector3.forward, m_Ortho.target);
        }

        // Look at a specific point from a given direction with a given zoom level, enabling and disabling perspective
        public void LookAt(Vector3 point, Quaternion direction, float newSize, bool ortho)
        {
            LookAt(point, direction, newSize, ortho, false);
        }

        // Look at a specific point from a given direction with a given zoom level, enabling and disabling perspective
        public void LookAt(Vector3 point, Quaternion direction, float newSize, bool ortho, bool instant)
        {
            ResetMotion();
            FixNegativeSize();

            if (instant)
            {
                m_Position.value = point;
                m_Rotation.value = direction;
                size = Mathf.Abs(newSize);
                m_Ortho.value = ortho;
                draggingLocked = DraggingLockedState.NotDragging;
            }
            else
            {
                m_Position.target = point;
                m_Rotation.target = direction;
                m_Size.target = Mathf.Abs(newSize);
                m_Ortho.target = ortho;
            }
            // Update name in the top-right handle
            // svRot.UpdateGizmoLabel(this, direction * Vector3.forward, m_Ortho.target);
        }
    }

    void OnGUI()
    {
        var r = GUILayoutUtility.GetRect(position.width, position.height);
        if (Event.current.type == EventType.Repaint)
        {
            m_SceneView.viewRect = r;
            preview.BeginPreview(r, GUIStyle.none);

            var camera = preview.camera;
            m_SceneView.camera = camera;
            camera.transform.rotation = m_SceneView.rotation;
            camera.transform.position = m_SceneView.pivot + camera.transform.rotation * new Vector3(0, 0, -m_SceneView.cameraDistance);

            camera.nearClipPlane = 0.1f;
            camera.farClipPlane = 100;

            preview.lights[0].intensity = 1.2f;
            preview.lights[0].transform.rotation = Quaternion.Euler(40, 40, 0);
            preview.lights[1].intensity = 1.2f;

            preview.DrawMesh(cubeMesh, Matrix4x4.identity, mat, 0);
            preview.camera.Render();
            var tex = preview.EndPreview();
            GUI.DrawTexture(r, tex, ScaleMode.StretchToFill, false);
            Repaint();
        }
        else
        {
            m_SceneView.UpdateCamera();
        }
    }
